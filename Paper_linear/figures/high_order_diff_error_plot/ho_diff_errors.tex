%%% START MACRO FOR ANNOTATION OF TRIANGLE WITH SLOPE %%%.
\newcommand{\logLogSlopeTriangle}[5]
{
    % #1. Relative offset in x direction.
    % #2. Relative offset in y direction.
    % #3. Slope d(y)/d(log10(x)).
    % #4. Plot options.
    \pgfplotsextra
    {
        \pgfkeys{/pgf/fpu}

        \pgfkeysgetvalue{/pgfplots/xmin}{\xmin}
        \pgfkeysgetvalue{/pgfplots/xmax}{\xmax}
        \pgfkeysgetvalue{/pgfplots/ymin}{\ymin}
        \pgfkeysgetvalue{/pgfplots/ymax}{\ymax}

        \pgfmathsetmacro{\triWidth}{exp(0.15*(\xmax-\xmin))}
        \pgfmathsetmacro{\triHeight}{\triWidth^#3}

        \pgfmathsetmacro{\xAAbs}{#1}
        \pgfmathsetmacro{\yAAbs}{#2}

        \pgfmathsetmacro{\xBAbs}{\xAAbs * \triWidth}
        \pgfmathsetmacro{\yBAbs}{\yAAbs}

        \pgfmathsetmacro{\xCAbs}{\xAAbs * \triWidth}
        \pgfmathsetmacro{\yCAbs}{\yAAbs * \triHeight}

        \pgfkeys{/pgf/fpu=false}

        \coordinate (A) at (axis cs: \xAAbs, \yAAbs);
        \coordinate (B) at (axis cs: \xBAbs, \yBAbs);
        \coordinate (C) at (axis cs: \xCAbs, \yCAbs);

        % Draw slope triangle.
        \draw[#4]   (A)-- node[pos=0.5,anchor=north] {\tiny1}
                    (B)-- node[midway,anchor=west] {\tiny#3}
                    (C)--
                    cycle;
    }
}
%%% END MACRO FOR ANNOTATION OF TRIANGLE WITH SLOPE %%%.
